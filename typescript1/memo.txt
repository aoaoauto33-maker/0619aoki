"devDependencies": {
  "tsx": "^4",　TypeScript本体
  "typescript": "^5",　　jsに変換するコマンドを打たずに実行できる便利なやつ
  "vite": "^6"　超高速な開発サーバー、ブラウザでの動作確認に使う
}必要なものをこういう風に書いておいて、npm installでインストールしてくれる
